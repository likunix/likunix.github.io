    JZ06.从尾到头打印链表

    【简单】输入一个链表的头节点，按链表从尾到头的顺序返回每个节点的值（用数组返回）。

    如：输入链表{1，2，3}：head->1->2->3

    返回数组 [3,2,1]，其中0<= 链表长度 <=10000

    /**
*    public class ListNode {
*        int val;
*        ListNode next = null;
*
*        ListNode(int val) {
*            this.val = val;
*        }
*    }
*/
//迭代
import java.util.*; 
public class Solution {
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
        ArrayList<Integer> list = new ArrayList<>();
        ListNode temp=listNode;
        while(temp!=null){
            list.add(0,temp.val);
            temp=temp.next;
        }
        return list;
    }
}
//递归
import java.util.*;
public class Solution{
    ArrayList<Integer> list = new ArrayList<>();
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode){
        if(listNode!=null){
            printListFromTailToHead(listNode.next);
            //递归调用语句后的语句执行顺序和各个被调用函数的顺序相反（出栈操作）
            list.add(listNode.val);
        }
        return list;
    }
}

    JZ24.反转链表

    【简单】给定一个单链表的头结点，长度为n，反转该链表后，返回新链表的表头，其中0≤n≤1000。

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//栈
import java.util.Stack;
public class Solution {
    public ListNode ReverseList(ListNode head) {
        Stack<ListNode> stack = new Stack<>();
        //链表各结点存栈
        while(head != null){
            stack.push(head);
            head = head.next;
        }
        //栈空说明链表为空，直接返回
        if(stack.isEmpty()){
            return null;
        }
        ListNode node = stack.pop();
        ListNode newHead = node;
        //结点出栈形成新链表
        while(!stack.isEmpty()){
            ListNode tempNode = stack.pop();
            node.next = tempNode;
            node = node.next;
        }
        //最后结点后继置空，防止成环
        node.next=null;
        return newHead;

    }
}
//迭代
//时间复杂度：O(n)，其中 n 是链表的长度。需要遍历链表一次。
//空间复杂度：O(1)
public class Solution {
    public ListNode ReverseList(ListNode head) {
        //处理空链表
        if(head==null)
            return head;
        ListNode pre = null;
        ListNode cur = head;
        while(cur != null){
            //断开链表，要记录后续一个
            ListNode temp = cur.next;
            //当前的next指向前一个
            cur.next = pre;
            //“前一个”更新为当前
            pre = cur;
            //"当前"更新为下一个
            cur = temp;
        }
        return pre;
    }
}

    JZ25.合并两个排序的链表

    【简单】输入两个递增的链表，单个链表长度为n，合并两个链表并且使新链表中的节点仍然是递增排序的，其中0≤n≤1000，−1000≤节点值≤1000。

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//递归
//时间复杂度：O(m+n)
//空间复杂度：O(m+n)
public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        //终止条件：两链表其中一个为空时，返回另一个链表
        if(list1==null)
            return list2;
        else if(list2==null)
            return list1;
        //比较两表头结点的值，值较小的list的头结点后面接merge好的链表（递归）
        if(list2.val>list1.val){
            list1.next = Merge(list1.next,list2);
            //每层返回选定的较小结点
            return list1;
        }
        else{
            list2.next = Merge(list1,list2.next);
            return list2;
        }
    }
}
//非递归
//时间复杂度：O(m+n)，m和n分别为两链表长度
//空间复杂度：O(1)
public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        ListNode dummy = new ListNode(-1);
        ListNode sentry = dummy;
        //特殊情况直接返回
        if(list1==null)
            return list2;
        if(list2==null)
            return list1;
        // 两个list都不为空时进入循环
        while(list1 != null & list2 != null) {
            if(list1.val > list2.val) {
                dummy.next = list2;
                list2 = list2.next;
                dummy = dummy.next;
            } else if(list1.val <= list2.val) {
                dummy.next = list1;
                list1 = list1.next;
                dummy = dummy.next;
            }
        }
        // list1后面还有，就把剩下的全部拿走
        if(list1 != null) {
            dummy.next = list1;
        }
        if(list2 != null) {
            dummy.next = list2;
        }
        return sentry.next;
    }
}

    JZ52.两个链表的第一个公共结点

    【简单】输入两个无环的单向链表，找出它们的第一个公共结点，如果没有公共节点则返回空。

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//差值法，利用公共节点前的节点不同的特点
public class Solution {
    public int length(ListNode phead){
        //计算输入链表长度
        ListNode p = phead;
        int len = 0;
        while(p != null){
            len++;
            p = p.next;
        }
        return len;
    }
    public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {
        int n1 = length(pHead1);
        int n2 = length(pHead2);
        if(n1>=n2){
            //链表1更长，链表1指针先走n1-n2步
            int n = n1 - n2;
            for(int i = 0;i < n ;i++){
                pHead1 = pHead1.next;
            }
            while((pHead1 != null)&&(pHead2 != null)&&(pHead1 != pHead2)){
                //两个链表同时移动，直到有公共结点时停下
                pHead1 = pHead1.next;
                pHead2 = pHead2.next;
            }
        }
        else{
            //反之，则链表2先行n2-n1步
            int n = n2-n1;
            for(int i = 0;i < n ;i++){
                pHead2 = pHead2.next;
            }
            while((pHead1 != null)&&(pHead2 != null)&&(pHead1 != pHead2)){
                //两个链表同时移动，直到有公共结点时停下
                pHead1 = pHead1.next;
                pHead2 = pHead2.next;
            }
        }
        return pHead1;
    }
}
//双指针遍历，利用两指针分别循环遍历两链表，一周期后停止在公共结点
public class Solution {
    public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {
        if(pHead1 == null ||pHead2 == null)
            //其中有一个为空，则不能有公共结点，返回null
            return null;
        ListNode p1 = pHead1;
        ListNode p2 = pHead2;
        while(p1 != p2){
            //遍历两个链表，遇到链表尾接上另一个的头
            p1 = p1 == null? pHead2 : p1.next;
            p2 = p2 == null? pHead1 : p2.next;
        }
        return p1;
    }
}

    JZ23.链表中环的入口结点

    【中等】给一个长度为n链表，若其中包含环，请找出该链表的环的入口结点，否则，返回null。

    1.判断有无环
    快指针步长为2，慢指针步长为1，相遇则说明有环
    2.有环找环的入口
    假设快指针在环中走了n圈，慢指针在环中走了m圈，它们才相遇，而进入环之前的距离为x，环入口到相遇点的距离为y，相遇点到环入口的距离为z。快指针一共走了x+n(y+z)+y步，慢指针一共走了x+m(y+z)+y，这个时候快指针走的倍数是慢指针的两倍，则x+n(y+z)+y=2(x+m(y+z)+y)，这时候x+y=(n−2m)(y+z)，因为环的大小是y+z，说明从链表头经过环入口到达相遇地方经过的距离等于整数倍环的大小

    用快慢指针判断有无环，利用 从链表头经过环入口到达相遇地方经过的距离等于整数倍环的大小

    /*
 public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}
*/
//快慢指针
public class Solution {
    public ListNode hasCycle(ListNode pHead){
        //先判断为空
        if(pHead == null)
            return null;
        //快慢指针
        ListNode fast =pHead;
        ListNode slow =pHead;
        //没环时快指针先到链表尾
        while(fast != null && fast.next != null){
            //快指针步长为2，慢指针步长为1，相遇则说明有环，返回相遇位置
            fast = fast.next.next;
            slow = slow.next;
            if(fast == slow)
                return slow;
        }
        return null;
    }
    public ListNode EntryNodeOfLoop(ListNode pHead) {
        //慢指针不动，快指针回到表头，步长都设置为1
        ListNode slow = hasCycle(pHead);
        if(slow == null)
            return null;
        ListNode fast = pHead;
        //再次相遇就是环的入口
        while(fast != slow){
            fast = fast.next;
            slow = slow.next;
        }
        return slow;
    }
}
//HashSet
//HashSet 是基于 HashMap 来实现的，是一个不允许有重复元素的集合。
//遍历链表中的每个结点，并将它记录下来；一旦遇到了此前遍历过的结点，就可以判定链表中存在环
import java.util.HashSet;
public class Solution {
    public ListNode EntryNodeOfLoop(ListNode pHead) {
        HashSet<ListNode> set = new HashSet<>();
        ListNode temp = pHead;
        while(temp != null){
            if(set.contains(temp)){
                遇到了此前遍历过的节点
                return temp;
            }
            set.add(temp);
            temp = temp.next;
        }
        return null;
    }
}

    另外：HashSet 去重功能是依赖元素的 hashCode 和 equals 方法判断的，通过这两个方法返回的都是 true 那就是相同对象，否则就是不同对象。

    JZ22.链表中倒数最后k个结点

    【简单】输入一个长度为 n 的链表，设链表中的元素的值为 ai ，返回该链表中倒数第k个节点。如果该链表长度小于k，请返回一个长度为 0 的链表。

    /*
 * public class ListNode {
 *   int val;
 *   ListNode next = null;
 *   public ListNode(int val) {
 *     this.val = val;
 *   }
 * }
 */
//一般方法
import java.util.*;
public class Solution {
    public ListNode FindKthToTail (ListNode pHead, int k) {
        // write code here
        int len = 0;
        ListNode p = pHead;
        while(p != null){
            遍历链表算长度
            len++;
            p = p.next;
        }
        if(len<k){
            链表过短直接返回
            return null;
        }
        p=pHead;
        for(int i=0;i<len-k;i++){
            遍历n-k次
            p = p.next;
        }
        return p;
}
//双指针
import java.util.*;
public class Solution {
    public ListNode FindKthToTail (ListNode pHead, int k) {
        ListNode fast = pHead;
        ListNode slow = pHead;
        //快指针先行k步
        for(int i=0;i<k;i++){
            if(fast != null){
                fast = fast.next;
            }
            else{
                //不够k步说明链表过短，直接返回
                return slow = null;
            }
        }
        //快慢指针同步前进，快指针到尾时，慢指针指向目标值
        while(fast != null){
            fast = fast.next;
            slow = slow.next;
        }
        return slow;
    }
}

    JZ35.复杂链表的复制

    【中等】请实现 copyRandomList 函数，复制一个复杂链表。在复杂链表中，每个节点除了有一个 next 指针指向下一个节点，还有一个 random 指针指向链表中的任意节点或者 null。

    /*
// Definition for a Node.
class Node {
    int val;
    Node next;
    Node random;

    public Node(int val) {
        this.val = val;
        this.next = null;
        this.random = null;
    }
}
*/
//哈希表
import java.util.Map;
import java.util.HashMap;
class Solution {
    public Node copyRandomList(Node head) {
        if(head == null) return null;
        Node cur = head;
        Map<Node, Node> map = new HashMap<>();
        // 3. 复制各节点，并建立 “原结点 -> 新结点” 的 Map 映射
        while(cur != null) {
            map.put(cur, new Node(cur.val));
            cur = cur.next;
        }
        cur = head;
        // 4. 构建新链表的 next 和 random 指向
        while(cur != null) {
            map.get(cur).next = map.get(cur.next);
            map.get(cur).random = map.get(cur.random);
            cur = cur.next;
        }
        // 5. 返回新链表的头结点
        return map.get(head);
    }
}

//拼接+拆分
//原节点 1 -> 新结点 1 -> 原结点 2 -> 新结点 2 -> ……
//难点：在访问原结点的 random 指向结点的同时找到新对应新结点的 random 指向结点
class Solution {
    public Node copyRandomList(Node head) {
        if(head == null) return null;
        Node cur = head;
        // 1. 复制各结点，并构建拼接链表
        while(cur != null) {
            Node tmp = new Node(cur.val);
            tmp.next = cur.next;
            cur.next = tmp;
            cur = tmp.next;
        }
        // 2. 构建各新结点的 random 指向
        //巧妙利用原链表的random指向和拼接链表的特性找到新链表的random指向
//--------------很巧妙------------------------------
        cur = head;
        while(cur != null) {
            if(cur.random != null)
                cur.next.random = cur.random.next;
            cur = cur.next.next;
        }
//-------------------------------------------------
        // 3. 拆分两链表
        cur = head.next;
        Node pre = head, res = head.next;
        while(cur.next != null) {
            pre.next = pre.next.next;
            cur.next = cur.next.next;
            pre = pre.next;
            cur = cur.next;
        }
        // 单独处理原链表尾结点
        pre.next = null;
        // 返回新链表头结点
        return res;      
    }
}

    JZ76.删除链表中重复的结点

    【中等】在一个排序的链表中，存在重复的结点，请删除该链表中重复的结点，重复的结点不保留，返回链表头指针。

    /*
 public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}
*/
//迭代法
import java.util.*;
public class Solution {
    public ListNode deleteDuplication(ListNode pHead) {
        //使用虚拟结点dummy,方便必要时删除首结点
        ListNode dummy = new ListNode(-1);
        ListNode tail = dummy;
        //以pHead为当前结点循环去掉重复的结点
        while (pHead != null){
            //如果当前结点为最后一个结点或与后面的值不同，则保留这个结点，接在tail后
            if(pHead.next == null || pHead.next.val != pHead.val){
                tail.next = pHead;
                tail = pHead;
            }
            //如果当前结点与下一结点相同，则跳过当前结点
            while(pHead.next != null && pHead.next.val == pHead.val){
                pHead = pHead.next;
            }
            pHead = pHead.next;
        }
        //对新链表尾结点next域置空
        tail.next = null;
        //返回新头结点
        return dummy.next;
    }
}

                pre.next = cur.next;
                cur = cur.next;
            }else{
                pre = cur;
                cur = cur.next;
            }
        }

    }

    JZ55.二叉树的深度

    【简单】输入一棵二叉树的根节点，求该树的深度。从根节点到叶节点依次经过的节点（含根、叶节点）形成树的一条路径，最长路径的长度为树的深度，根节点的深度视为 1 。

    DFS：从图中一个未访问的顶点 V 开始，沿着一条路一直走到底，然后从这条路尽头的节点回退到上一个节点，再从另一条路开始走到底…，不断递归重复此过程，直到所有的顶点都遍历完成。

    树是图的一种特例(连通无环的图就是树)，树的前序遍历，中序遍历，后序遍历，都属于深度优先遍历

    实现方式有递归和非递归，递归方式通过不断递归处理当前结点的左右子树，树的深度过高可能会导致栈溢出；非递归方式遍历时可以使用栈来压入当前结点的右、左结点，然后弹出来非空的话重复以上步骤来遍历

    BFS：从图的一个未遍历的节点出发，先遍历这个节点的相邻节点，再依次遍历每个相邻节点的相邻节点。

    广度优先遍历用队列来实现

    /**
*public class TreeNode {
*    int val;
*    TreeNode left;
*    TreeNode right;
*    TreeNode(int x){
*        this.val = x;
*    }
*} 
*/
//DFS 递归
class Solution {
    public int maxDepth(TreeNode root) {
        if(root==null){
            return 0;
        }else{
            //递归求根节点左子树深度
            int leftHeight = maxDepth(root.left);
            //递归求根节点右子树深度
            int rightHeight = maxDepth(root.right);
            //返回左右子树之间深度较大的那个加上根节点深度1
            return Math.max(leftHeight,rightHeight)+1;
        }
    }
}
//BFS
//树的层序遍历 / 广度优先搜索往往利用 队列 实现。每遍历一层，则计数器 +1，直到遍历完成，则可得到树的深度
class Solution {
    public int maxDepth(TreeNode root) {
        if(root==null){
            return 0;
        }
        //队列queue存当前层结点，temp存下一层结点
        Queue<TreeNode> queue = new LinkedList<>();
        Queue<TreeNode> temp;
        queue.add(root);
        int res = 0;
        while(!queue.isEmpty()){
            temp = new LinkedList<>();
            for(TreeNode node : queue){
                //对于当前层的每个结点，如果不是叶子结点，就将子结点加入队列temp
                if(node.left!=null){
                    temp.add(node.left);
                }
                if(node.right!=null){
                    temp.add(node.right);
                } 
            }
            //更新队列queue值为下一层结点
            queue = temp;
            //层数+1
            res++;
        }
        return res;
    }
}

            return res;

                TreeNode node = s1.pop();
                temp.add(node.val);
                //子结点加入偶数层
                if(node.left != null){
                    s2.push(node.left);
                }
                if(node.right != null){
                    s2.push(node.right);
                }
            }
            if(temp.size()!=0){
              //不直接res.add(temp);应该是新开辟空间，避免变量内存被修改，导致引用类型也跟着被修改
                res.add(new ArrayList<Integer>(temp));
            }

            temp.clear();
            while(!s2.isEmpty()){
                //记录偶数层
                TreeNode node = s2.pop();
                temp.add(node.val);
                //子结点加入奇数层
                if(node.right != null){
                    s1.push(node.right);
                }
                if(node.left != null){
                    s1.push(node.left);
                }
            }
            //数组不为空时才添加
            if(temp.size()!= 0){
                res.add(new ArrayList<Integer>(temp));
            }
            //清空本层数据
            temp.clear();
        }

class Solution {
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        if (preorder == null || preorder.length == 0) {
            return null;
        }
        TreeNode root = new TreeNode(preorder[0]);
        Deque<TreeNode> stack = new LinkedList<TreeNode>();
        stack.push(root);
        int inorderIndex = 0;
        for (int i = 1; i < preorder.length; i++) {
            int preorderVal = preorder[i];
            TreeNode node = stack.peek();
            if (node.val != inorder[inorderIndex]) {
                node.left = new TreeNode(preorderVal);
                stack.push(node.left);
            } else {
                while (!stack.isEmpty() && stack.peek().val == inorder[inorderIndex]{
                    node = stack.pop();
                    inorderIndex++;
                }
                node.right = new TreeNode(preorderVal);
                stack.push(node.right);
            }
        }
        return root;
    }
}

    public boolean isSubStructure(TreeNode A, TreeNode B) {
        //约定空树不是任意一个树的子结构
        if(A==null || B==null) return false;
        //在A树中查找与B树一样的子结构

        //否则，继续往下比较
        return isSame(t1.left, t2.left) && isSame(t1.right, t2.right);
    }

    【简单】定义栈的数据结构，请在该类型中实现一个能够得到栈中所含最小元素的 min 函数，输入操作时保证 pop、top 和 min 函数操作时，栈中一定有元素。

    此栈包含的方法有：
    push(value):将value压入栈中
    pop():弹出栈顶元素
    top():获取栈顶元素
    min():获取栈中最小元素

    import java.util.Stack;
//双栈
public class Solution {
    //记录压栈元素
    Stack<Integer> s1 = new Stack<>();
    //记录当前最小值
    Stack<Integer> s2 = new Stack<>();
    public void push(int node) {
        s1.push(node);
        //比较压栈元素与最小值的关系
        if(s2.isEmpty() || s2.peek()>node){
            s2.push(node);
        }else{
            //不符合说明压栈后，s2的顶还是当前的最小值，
            //重复压一次s2栈顶元素使每次出栈后s2都记录着当前的最小值
            s2.push(s2.peek());
        }
    }
    public void pop() {
        s1.pop();
        s2.pop();
    }
    public int top() {
        return s1.peek();
    }
    public int min() {
        return s2.peek();
    }
}
//ArrayList
public class Solution {
    //记录压栈元素
    Stack<Integer> stack = new Stack<>();
    //用于获取最小值
    ArrayList<Integer> list = new ArrayList<>();
    public void push(int node) {
        stack.push(node);
        list.add(node);
    }
    public void pop() {
        list.remove(stack.pop());
    }
    public int top() {
        return stack.peek();
    }
    public int min() {
        //先调用Collections工具类，的静态方法sort对list进行排序再返回
        Collections.sort(list);
        return list.get(0);
    }
}

    JZ31.栈的压入、弹出序列

    【中等】输入两个整数序列，第一个序列表示栈的压入顺序，请判断第二个序列是否可能为该栈的弹出顺序。假设压入栈的所有数字均不相等。

    例如序列1,2,3,4,5是某栈的压入顺序，序列4,5,3,2,1是该压栈序列对应的一个弹出序列，但4,3,5,1,2就不可能是该压栈序列的弹出序列

    1. 0<=pushV.length == popV.length <=1000
    2. -1000<=pushV[i]<=1000
    3. pushV 的所有数字均不相同
    //模拟法
class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        Stack<Integer> s = new Stack<>();
        int n = 0;
        //根据入栈序列入栈
        for(int num : pushed){
            s.push(num);
            //循环比较，栈顶元素等于出栈序列就出栈，比较下一个        
            while(!s.isEmpty() && s.peek()==popped[n]){
                s.pop();
                n++;
            }
        }
        //栈非空说明可以产生该出栈序列
        return s.isEmpty();
    }
}
//原地栈 跟方法一一样，不过利用了数组pushed前面部分入栈后便不再使用的空间来作为“栈”
class Solution {
    public boolean validateStackSequences(int [] pushed ,int [] popped {
        //n表示栈大小，并且pushed[n]表示栈顶元素
        int n = 0;
        //i表示popped[i]的下标
        int i = 0;
        for(int num : pushed){
            //注意下面这行代码不能省略，因为pushed[n]代表栈顶，而for循环不再前进让n++时，
            //后面while循环可能会使栈顶n减小,此时压栈内容不等于数组pushed的第n个元素
            pushed[n] = num;
            //--------------
            while(n>=0 && pushed[n]==popped[i]){
                n--;
                i++;
            }
            n++;
        } 
        //n==0表示“栈为空”
        return n==0;
    }
}

class Solution {
    public String reverseWords(String str) {
        str = str.trim();
        int tail = str.length()-1,head = tail;
        StringBuilder res = new StringBuilder();
        while(head >= 0){
            while(head >= 0 && str.charAt(head)!=' ')
                head--;
            res.append(str.substring(head+1,tail+1)+" ");
            while(head>=0 && str.charAt(head)==' ')
                head--;
            tail=head;
        }
        return res.toString().trim();
    }
}

    JZ59.滑动窗口的最大值

    【中等】给定一个数组 nums 和滑动窗口的大小 k，请找出所有滑动窗口里的最大值。

    //第一次做想到的解法，时间复杂度较高
class Solution {
    public int maxInNumsFromAToB(int[] nums,int a,int b){
        //求nums数组中a到b这个窗口内的最大值
        int max = 0;
        for(int i =a;i<=b;i++){
            if(nums[i]>=max){
                max = nums[i];
            }
        }
        return max;
    }
    public int[] maxSlidingWindow(int[] nums, int k) {
        //数组为空或者窗口大小为1，直接返回原数组nums的引用
        if(nums.length==0 || k==1) {
            return nums;
        }
        //滑动窗口的个数
        int len = nums.length-k+1;
        int[] res = new int[len];
        int a = 0;
        int b = k-1;
        for(int i=0;i<len;i++){ 
            //窗口移动
            res[i]=maxInNumsFromAToB(nums,a,b);
            a++;
            b++;
        }
        return res;
    }
}
//单调栈
//由于滑动窗口移动时，出栈操作要在列表尾部删除，故适合用双端队列
class Solution {
    public int[] maxSlidingWindow(int[] nums,int k){
        if(nums.length==0 || k==0){
            return new int[0];
        }
        //双端队列
        Deque<Integer> deque = new LinkedList<>();
        int[] res = new int[nums.length-k+1];
        for(int j = 0,i=1-k;j<nums.length;i++,j++){
            //删除deque中对应的nums[i-1]
            if(i>0 && deque.peekFirst()==nums[i-1])
                deque.removeFirst();
            //保持deque递减
            while(!deque.isEmpty() && deque.peekLast()<nums[j])
                deque.removeLast();
            deque.addLast(nums[j]);
            //记录窗口最大值
            if(i>=0){
                res[i] = deque.peekFirst();
            }
        }
        return res;
    }
}
