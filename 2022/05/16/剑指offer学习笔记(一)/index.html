    JZ06.从尾到头打印链表

    【简单】输入一个链表的头节点，按链表从尾到头的顺序返回每个节点的值（用数组返回）

    如：输入链表{1，2，3}：head->1->2->3

    返回数组 [3,2,1]，其中0<= 链表长度 <=10000

    题解：

    /**
*    public class ListNode {
*        int val;
*        ListNode next = null;
*
*        ListNode(int val) {
*            this.val = val;
*        }
*    }
*/
//迭代
import java.util.*; 
public class Solution {
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
        ArrayList<Integer> list = new ArrayList<>();
        ListNode temp=listNode;
        while(temp!=null){
            list.add(0,temp.val);
            temp=temp.next;
        }
        return list;
    }
}
//递归
import java.util.*;
public class Solution{
    ArrayList<Integer> list = new ArrayList<>();
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode){
        if(listNode!=null){
            printListFromTailToHead(listNode.next);
            //递归调用语句后的语句执行顺序和各个被调用函数的顺序相反（出栈操作）
            list.add(listNode.val);
        }
        return list;
    }
}

    JZ24.反转链表

    【简单】给定一个单链表的头结点，长度为n，反转该链表后，返回新链表的表头，其中0≤n≤1000

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//栈
import java.util.Stack;
public class Solution {
    public ListNode ReverseList(ListNode head) {
        Stack<ListNode> stack = new Stack<>();
        //链表各节点存栈
        while(head != null){
            stack.push(head);
            head = head.next;
        }
        //栈空说明链表为空，直接返回
        if(stack.isEmpty()){
            return null;
        }
        ListNode node = stack.pop();
        ListNode newHead = node;
        //节点出栈形成新链表
        while(!stack.isEmpty()){
            ListNode tempNode = stack.pop();
            node.next = tempNode;
            node = node.next;
        }
        //最后节点后继置空，防止成环
        node.next=null;
        return newHead;

    }
}
//迭代
//时间复杂度：O(n)，其中 n 是链表的长度。需要遍历链表一次。
//空间复杂度：O(1)
public class Solution {
    public ListNode ReverseList(ListNode head) {
        //处理空链表
        if(head==null)
            return head;
        ListNode pre = null;
        ListNode cur = head;
        while(cur != null){
            //断开链表，要记录后续一个
            ListNode temp = cur.next;
            //当前的next指向前一个
            cur.next = pre;
            //“前一个”更新为当前
            pre = cur;
            //"当前"更新为下一个
            cur = temp;
        }
        return pre;
    }
}

    JZ25.合并两个排序的链表

    【简单】输入两个递增的链表，单个链表长度为n，合并两个链表并且使新链表中的节点仍然是递增排序的，其中0≤n≤1000，−1000≤节点值≤1000

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//递归
//时间复杂度：O(m+n)
//空间复杂度：O(m+n)
public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        //终止条件：两链表其中一个为空时，返回另一个链表
        if(list1==null)
            return list2;
        else if(list2==null)
            return list1;
        //比较两表头结点的值，值较小的list的头结点后面接merge好的链表（递归）
        if(list2.val>list1.val){
            list1.next = Merge(list1.next,list2);
            //每层返回选定的较小结点
            return list1;
        }
        else{
            list2.next = Merge(list1,list2.next);
            return list2;
        }
    }
}
//非递归
//时间复杂度：O(m+n)，m和n分别为两链表长度
//空间复杂度：O(1)
public class Solution {
    public ListNode Merge(ListNode list1,ListNode list2) {
        ListNode dummy = new ListNode(-1);
        ListNode sentry = dummy;
        //特殊情况直接返回
        if(list1==null)
            return list2;
        if(list2==null)
            return list1;
        // 两个list都不为空时进入循环
        while(list1 != null & list2 != null) {
            if(list1.val > list2.val) {
                dummy.next = list2;
                list2 = list2.next;
                dummy = dummy.next;
            } else if(list1.val <= list2.val) {
                dummy.next = list1;
                list1 = list1.next;
                dummy = dummy.next;
            }
        }
        // list1后面还有，就把剩下的全部拿走
        if(list1 != null) {
            dummy.next = list1;
        }
        if(list2 != null) {
            dummy.next = list2;
        }
        return sentry.next;
    }
}

    JZ52.两个链表的第一个公共结点

    【简单】输入两个无环的单向链表，找出它们的第一个公共结点，如果没有公共节点则返回空

    /*
public class ListNode {
    int val;
    ListNode next = null;

    ListNode(int val) {
        this.val = val;
    }
}*/
//差值法，利用公共节点前的节点不同的特点
public class Solution {
    public int length(ListNode phead){
        //计算输入链表长度
        ListNode p = phead;
        int len = 0;
        while(p != null){
            len++;
            p = p.next;
        }
        return len;
    }
    public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {
        int n1 = length(pHead1);
        int n2 = length(pHead2);
        if(n1>=n2){
            //链表1更长，链表1指针先走n1-n2步
            int n = n1 - n2;
            for(int i = 0;i < n ;i++){
                pHead1 = pHead1.next;
            }
            while((pHead1 != null)&&(pHead2 != null)&&(pHead1 != pHead2)){
                //两个链表同时移动，直到有公共结点时停下
                pHead1 = pHead1.next;
                pHead2 = pHead2.next;
            }
        }
        else{
            //反之，则链表2先行n2-n1步
            int n = n2-n1;
            for(int i = 0;i < n ;i++){
                pHead2 = pHead2.next;
            }
            while((pHead1 != null)&&(pHead2 != null)&&(pHead1 != pHead2)){
                //两个链表同时移动，直到有公共结点时停下
                pHead1 = pHead1.next;
                pHead2 = pHead2.next;
            }
        }
        return pHead1;
    }
}
//双指针遍历，利用两指针分别循环遍历两链表，一周期后停止在公共结点
public class Solution {
    public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {
        if(pHead1 == null ||pHead2 == null)
            //其中有一个为空，则不能有公共结点，返回null
            return null;
        ListNode p1 = pHead1;
        ListNode p2 = pHead2;
        while(p1 != p2){
            //遍历两个链表，遇到链表尾接上另一个的头
            p1 = p1 == null? pHead2 : p1.next;
            p2 = p2 == null? pHead1 : p2.next;
        }
        return p1;
    }
}
